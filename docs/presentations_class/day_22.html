Loading Packages

library(trelliscopejs)
library(tidyverse)
data(permits, package="buildings")

Nesting Data

#by state and county

by_stco <- permits %>%
  filter(StateAbbr %in% c("WA", "ID", "UT"), variable == "Single Family") %>%
  group_by(StateAbbr, countyname) %>%
  nest()

Adding Plots

by_stco <- by_stco %>% 
  mutate(
    panel = map_plot(data,
                     ~ ggplot(data = .x, aes(x = year, y = value)) +
                       geom_point() +
                       geom_line(color = "grey") +
                       #xlim(2000, 2011) + 
                       #ylim(0, 1250) +
                       theme_bw()
    ))

Making the Magic

# plot it
my_display <- tempfile()
by_stco %>%
  trelliscope("permits", nrow = 2, ncol = 7, 
              path = my_display, thumb = TRUE, width = 500)
